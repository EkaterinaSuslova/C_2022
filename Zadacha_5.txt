#include <stdio.h>
#include <iostream>

using namespace std ;


float SquaredSum (float numberOne, float numberTwo) {
  return (numberOne * numberOne) + (numberTwo * numberTwo);
}
void Sort(int elm[], int len) {

  for (int x = 0; x < len - 1; x++) {
    for (int k = x + 1; k < len; k++) {
      if (elm[x] < elm[k]) {
        int a = elm[k];
        elm[k] = elm[x];
        elm[x] = a;
 }
 }
 }
 }
